玩客云openwrt稳定高大全设置静IP

2023年09月27日 10:00:00 | 代码命令

百度网盘下载地址

##刷机过程不演示，自已百度查找方法

#玩客云高大全直刷，全分区去校验，，登陆帐号：root 密码为：root，主路由ip10.0.0.1需网线直连电脑配置旁路由IP，用自带的一键设置就行，
建议:刷主路由固件，直连电脑重新在管理页面配置旁路由

ssh root@192.168.3.34

vim /etc/config/network

这时你会看到出现
config interface 'loopback'
       option device 'lo'
       option proto 'static'
       option ipaddr '127.0.0.1'
       option netmask '255.0.0.0'

config globals 'globals'
       option ula_prefix 'fd6b:6253:662c::/48'
       option packet_steering '1'

config device
       option name 'br-lan'
       option type 'bridge'
       list ports 'eth0'

config interface 'lan'
       option device 'br-lan'
       option proto 'dhcp'

config interface 'vpn0'
       option device 'tun0'
       option proto 'none'

config interface 'ipsec_server'
       option device 'ipsec0'
       option proto 'static'
       option ipaddr '192.168.100.1'
-----------------------------------------
按英文字母i可以进入修改模式，用第三方软件例如：FinalShell 可以复制粘贴的,我的修改就是把MAC地址固定于1A:AA:A5:61:7B:7D，其它没什么改动的
改完后按键盘Esc键 ，再按Shift加：号键，再你w-q回车，退出并保存

       我的修改成这下面这样
       config interface 'loopback'
       option proto 'static'
       option ipaddr '127.0.0.1'
       option netmask '255.0.0.0'
       option device 'lo'
       option macaddr '1A:AA:A5:61:7B:7D'

config globals 'globals'
       option ula_prefix 'fd7b:ad35:5cfd::/48'
       option macaddr '1A:AA:A5:61:7B:7D'

config interface 'lan'
       option proto 'dhcp'
       option device 'br-lan'
       option macaddr '1A:AA:A5:61:7B:7D'

config interface 'docker'
       option device 'docker0'
       option proto 'none'
       option auto '0'
       option macaddr '1A:AA:A5:61:7B:7D'

config device
       option type 'bridge'
       option name 'docker0'
       option macaddr '1A:AA:A5:61:7B:7D'
       
       然后重启网络
```		
/etc/init.d/network restart
```			
这时登陆路由器看MAc：1A:AA:A5:61:7B:7D的设备就上线了，你这个对应路由器的IP地址也会永远固定这个，如果你想改变IP地址，就在主路由上设置，一般路由器都是找到网络设置，静态 IP 地址绑定列表，找到1A:AA:A5:61:7B:7D这个对应的名称，绑定就可以，因为这个玩客云路由器不能用静态功能，我试过，每次用静态功能就上不了网，例如科学上网等等！！！
